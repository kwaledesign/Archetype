Link Styles

/**
 * Links
 *
 * Links are scoped with $main-type-scope which is set in the base/_typography.scss partial
 */

.#{$main-type-scope} {
  a {
    color: blue;
    font-weight: normal;                   
    text-decoration: none;
    &:visited { 
      color: blue; 
    }
    &:hover { 
      color: blue; 
    } 
    &:focus { 
      color: blue; 
    }
  }
}
