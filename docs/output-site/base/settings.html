Settings

The _settings.scss partial contains both Compass and Archetype settings that are applied project wide. Items such as vertical-rhythm and legacy browser support as well as Archetype's implementation of Normalize are all controlled here.

/**
 * Archetype Settings 
 *
 */


/**
 * Imports
 *
 * 1. https://github.com/chriseppstein/compass
 * 2. https://github.com/kwaledesign/Archetype
 */

@import "compass";                                        /* 1 */
@import "archetype";                                      /* 2 */

/**
 * Compass
 * 
 * 1. Enable Compass to use HTML hacks to support IE 6 and 7
 *    Note: Archetype supports IE9+ (IE6-8 are served a linear layout)
 */
$legacy-support-for-ie6 : false;                          /* 1 */
$legacy-support-for-ie7 : false;                          /* 1 */
$legacy-support-for-ie8 : true;                           /* 1 */


/**
 * Normalize Mixin
 *
 * https://github.com/kwaledesign/Archetype/blob/master/stylesheets/archetype/_normalize.scss
 * 
 * comment out any unneeded components
 */
@mixin normalize() {
  @include normalize-display;
  @include normalize-page;
  @include normalize-links;
  @include normalize-typography;
  @include normalize-embeds;
  @include normalize-figures;
  @include normalize-forms;
  @include normalize-tables;
}
/* call normalize mixin */
@include normalize;
